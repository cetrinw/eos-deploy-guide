# cleos shell

cleos get account prodsadmin

# 解除质押资源
cleos  system undelegatebw prodsadmin prodsadmin "9900000 SYS" "9 SYS" -p prodsadmin@active

cleos system newaccount accountadmin aaaaaaaaaaav  --transfer EOS542apSbR2jPHqKKSqPJxs9J2zsmh3ioiMsPC4sZV48wKQhMe2C EOS542apSbR2jPHqKKSqPJxs9J2zsmh3ioiMsPC4sZV48wKQhMe2C --stake-net "20000.0000 SYS" --stake-cpu "20000.0000 SYS" --buy-ram-kbytes 4000000 -p accountadmin@active
